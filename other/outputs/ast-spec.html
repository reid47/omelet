AST Specification

Note: By convention, any property value that starts with a capital letter is an AST node. Capital letters should only be used for the names of nodes.

Array

kind: a string literal, "Array"
start: a number literal
end: a number literal

values: [Number | String | Boolean | Tag]

Assignment

kind: a string literal, "Assignment"
start: a number literal
end: a number literal

leftSide: Identifier
rightSide: an expression

Attribute

kind: a string literal, "Attribute"
start: a number literal
end: a number literal

name: String
value: String | Interpolation

Boolean

kind: a string literal, "Boolean"
start: a number literal
end: a number literal

value: a boolean literal

Comment

kind: a string literal, "Comment"
start: a number literal
end: a number literal

value: a string literal

CommentHTML

kind: a string literal, "CommentHTML"
start: a number literal
end: a number literal

value: a string literal

Doctype

kind: a string literal, "Doctype"
start: a number literal
end: a number literal

value: a string literal

Document

kind: a string literal, "Document"
start: a number literal
end: a number literal

extend: Extend
imports: [Import]
contents: [MacroDefinition | Assignment | an expression]

Extend

kind: a string literal, "Extend"
start: a number literal
end: a number literal

file: a string literal

Filter

kind: a string literal, "Filter"
start: a number literal
end: a number literal

name: Identifier
arguments: [Number | String | Identifier]

ForEach

kind: a string literal, "ForEach"
start: a number literal
end: a number literal

iterator: Identifier i.e. item in (for item in items :: ... )
data: Identifier i.e. items in (for item in items :: ... )
body: [an expression]

Identifier

kind: a string literal, "Identifier"
start: a number literal
end: a number literal

value: a string literal
modifiers: undefined | [Modifier]

IfStatement

kind: a string literal, "IfStatement"
start: a number literal
end: a number literal

predicate: Boolean | Parenthetical | Interpolation
thenCase: [an expression]
elifCases: undefined | [IfStatement] turned into IfStatements without elifCases or elseCases
elseCase: undefined | [an expression]

Import

kind: a string literal, "Import"
start: a number literal
end: a number literal

file: a string literal

Include

kind: a string literal, "Include"
start: a number literal
end: a number literal

file: a string literal

Interpolation

kind: a string literal, "Interpolation"
start: a number literal
end: a number literal

name: Identifier
arguments: [Number | Boolean | String | Identifier]
filters: [Filter]

MacroDefinition

kind: a string literal, "MacroDefinition"
start: a number literal
end: a number literal

name: Identifier
params: [Identifier]
body: a single expression

Modifier

kind: a string literal, "Modifier"
start: a number literal
end: a number literal

value: Number | Identifier i.e. 3 in items[3] or color in car.color

Number

kind: a string literal, "Number"
start: a number literal
end: a number literal

value: a string literal

Parenthetical

kind: a string literal, "Parenthetical"
start: a number literal
end: a number literal

inner: [expressions]
filters: [Filter]

Raw

kind: a string literal, "Raw"
start: a number literal
end: a number literal

value: a string literal

String

kind: a string literal, "String"
start: a number literal
end: a number literal

value: a string literal

Symbol

kind: a string literal, "Symbol"
start: a number literal
end: a number literal

value: a string literal

Tag

kind: a string literal, "Tag"
start: a number literal
end: a number literal

name: String
attributes: [Attribute] class attributes are merged into one before added to AST
inner: [an expression]
filters: undefined | [Filter]
